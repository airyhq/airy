# Airy Chat Plugin 

## Install

```bash
npm install --save @airyhq/chat-plugin
```

or

```bash
yarn add @airyhq/chat-plugin
``` 

## Usage

```tsx
import React from 'react'
import {AiryChatPlugin, AiryChatPluginConfiguration} from 'chat-plugin';

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
